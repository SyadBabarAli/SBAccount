CREATE FUNCTION [dbo].[VGC1](@tableName varchar(50))
RETURNS TABLE
AS 
RETURN
--Data
SELECT 
COLUMN_NAME
,CASE 

WHEN  (ORDINAL_POSITION = 1 OR ORDINAL_POSITION = 2)  AND DATA_TYPE = 'INT' AND IS_NULLABLE = 'NO'  
THEN '[Key, Column(Order = '+CONVERT(VARCHAR,ORDINAL_POSITION)+')] public int '+COLUMN_NAME+' { get; set; }'
WHEN DATA_TYPE = 'varchar' OR DATA_TYPE = 'nchar' OR DATA_TYPE = 'nvarchar' --AND IS_NULLABLE = 'YES' 
THEN 'public string '+COLUMN_NAME+' { get; set; }'
WHEN DATA_TYPE = 'bit' AND IS_NULLABLE = 'YES' 
THEN 'public bool? '+COLUMN_NAME+' { get; set; }'
WHEN DATA_TYPE = 'decimal' AND IS_NULLABLE = 'YES' 
THEN 'public decimal? '+COLUMN_NAME+' { get; set; }'
WHEN DATA_TYPE = 'decimal' AND IS_NULLABLE = 'NO' 
THEN 'public decimal '+COLUMN_NAME+' { get; set; }'
WHEN DATA_TYPE = 'bit' AND IS_NULLABLE = 'NO' 
THEN 'public bool '+COLUMN_NAME+' { get; set; }'
WHEN DATA_TYPE = 'int' AND IS_NULLABLE = 'YES' 
THEN 'public int? '+COLUMN_NAME+' { get; set; }'
WHEN DATA_TYPE = 'int' AND IS_NULLABLE = 'NO' 
THEN 'public int '+COLUMN_NAME+' { get; set; }'
WHEN DATA_TYPE = 'datetime' OR DATA_TYPE = 'date' AND IS_NULLABLE = 'YES' 
THEN 'public DateTime? '+COLUMN_NAME+' { get; set; }'
WHEN DATA_TYPE = 'datetime' OR DATA_TYPE = 'date' AND IS_NULLABLE = 'NO' 
THEN 'public DateTime '+COLUMN_NAME+' { get; set; }'
END 'Model'
,CASE 

WHEN  (ORDINAL_POSITION = 1 OR ORDINAL_POSITION = 2)  AND DATA_TYPE = 'INT' AND IS_NULLABLE = 'NO'  THEN 
' <v-layout> <v-flex lg3> <input readonly  v-model="editedItem.'+COLUMN_NAME+'" label="'+COLUMN_NAME+'" style="display:none;" /> </v-flex></v-layout> '
WHEN  DATA_TYPE = 'INT' AND CHARINDEX('id', COLUMN_NAME) > 0 THEN 
'<v-layout><v-flex lg3>'+
'<v-combobox v-model="select'+REPLACE(COLUMN_NAME,'ID','')+'" '+
' :items="items'+REPLACE(COLUMN_NAME,'ID','')+'"'+
' label="'+COLUMN_NAME+'"'+
' item-text="text"'+
' item-value="vaue">'+
' </v-combobox>'+
' </v-flex> </v-layout>'
WHEN  DATA_TYPE = 'BIT' THEN
'<v-layout> <v-flex lg3><v-checkbox color="colorDark" v-model="editedItem.'+COLUMN_NAME+'" label="'+COLUMN_NAME+'" ></v-checkbox>  </v-flex> </v-layout>'
WHEN  DATA_TYPE = 'VARCHAR' THEN 
'<v-layout><v-flex lg3><v-text-field placeholder="Placeholder" v-model="editedItem.'+COLUMN_NAME+'" label="'+COLUMN_NAME+'"></v-text-field> </v-flex> </v-layout>' 
WHEN  DATA_TYPE = 'DECIMAL' THEN 
'<v-layout><v-flex lg3><v-text-field placeholder="Placeholder" v-model="editedItem.'+COLUMN_NAME+'" label="'+COLUMN_NAME+'"></v-text-field> </v-flex> </v-layout>' 
WHEN  DATA_TYPE = 'INT' THEN 
'<v-layout><v-flex lg3><v-text-field placeholder="Placeholder" v-model="editedItem.'+COLUMN_NAME+'" label="'+COLUMN_NAME+'"></v-text-field> </v-flex> </v-layout>' 
WHEN  DATA_TYPE = 'DATETIME' THEN 
'<v-layout><v-flex lg3><v-text-field label="'+COLUMN_NAME+'" placeholder="Placeholder" :append-icon="today"type="date" :value="'+COLUMN_NAME+' && '+COLUMN_NAME+'.toISOString().split("T")[0]" ></v-text-field> </v-flex> </v-layout>' 
END  'DesignHTML'
 ,'{ text: "'+COLUMN_NAME++'", value: "'+COLUMN_NAME++'" },' 'Header'
,CASE WHEN CHARINDEX('id', COLUMN_NAME) > 0 THEN 'select'+REPLACE(COLUMN_NAME,'ID','')+': [],'+'items'+REPLACE(COLUMN_NAME,'ID','')+': [],' ELSE (CASE WHEN DATA_TYPE ='Bit' THEN  COLUMN_NAME+': true,' ELSE  COLUMN_NAME+': "",' END) END  'ClientData'

,CASE WHEN CHARINDEX('id', COLUMN_NAME) > 0 THEN 'obj.'+COLUMN_NAME +' = this.select'+REPLACE(COLUMN_NAME,'ID','')+'.value' ELSE  'obj.'+COLUMN_NAME +'=this.editedItem.'+ COLUMN_NAME +';' END 'PostClient'
,'<td class="text-xs-left">{{ props.item.'+COLUMN_NAME+' }}</td>' 'HeaderVUE'

,COLUMN_NAME+' = t1.'+COLUMN_NAME+',' 'Select'
, 'obj.'+COLUMN_NAME +'= objModel.'+ COLUMN_NAME+';' 'Post'


--ELSE 'public '+ (CASE WHEN DATA_TYPE = 'bit' THEN 'bool?' ELSE DATA_TYPE END ) +' '+COLUMN_NAME+' { get; set; }' END 'Model'

--'public '+ CASE WHEN DATA_TYPE = 'varchar' then 'string' else DATA_TYPE end +' '+COLUMN_NAME+' { get; set; }' 'Model'

, 'this.editedItem.'+COLUMN_NAME +'= obj.'+ COLUMN_NAME+';' 'editItem'
,DATA_TYPE
FROM INFORMATION_SCHEMA.COLUMNS WHERE TABLE_NAME = @tableName 
--AND COLUMN_NAME NOT IN('Created','Modified','CreatedBy','ModifiedBy','CompanyId',@tableName+'Id')



--SELECT * FROM INFORMATION_SCHEMA.COLUMNS WHERE TABLE_NAME ='ListProduct'
GO
